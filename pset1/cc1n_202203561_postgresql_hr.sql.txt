Essa é a segunda sessão do código, onde está os comandos para criação de tabelas, com seus devidos comentários de tabela e

CREATE TABLE hr.cargos (
                id_cargo VARCHAR(10) NOT NULL,
                cargo VARCHAR(35) NOT NULL,
                salario_minimo NUMERIC(8,2),
                salario_máximo NUMERIC(8,2),
                CONSTRAINT id_cargo PRIMARY KEY (id_cargo)
);
COMMENT ON TABLE hr.cargos IS 'Tabela cargos, que armazena os cargos existentes na empresa, e o salário mínimo e máximo de cada cargo.';
COMMENT ON COLUMN hr.cargos.id_cargo IS 'Chave primária da tabela cargos.';
COMMENT ON COLUMN hr.cargos.cargo IS 'Nome do cargo.';
COMMENT ON COLUMN hr.cargos.salario_minimo IS 'O menor salário mínimo admitido para um cargo.';
COMMENT ON COLUMN hr.cargos.salario_máximo IS 'O maior salário admitido para um cargo.';


CREATE UNIQUE INDEX cargos_idx
 ON hr.cargos
 ( cargo );

CREATE TABLE hr.regioes (
                id_regiao INTEGER NOT NULL,
                nome VARCHAR(25) NOT NULL,
                CONSTRAINT id_regiao PRIMARY KEY (id_regiao)
);
COMMENT ON TABLE hr.regioes IS 'Tabela de reigões. Contém os identificadores e o nome das regiões.';
COMMENT ON COLUMN hr.regioes.id_regiao IS 'Chave primária da tabela regiões.';
COMMENT ON COLUMN hr.regioes.nome IS 'Nome da região contida na tabela.';


CREATE UNIQUE INDEX regioes_idx
 ON hr.regioes
 ( nome );

CREATE TABLE hr.paises (
                id_pais CHAR(2) NOT NULL,
                nome VARCHAR(50) NOT NULL,
                id_regiao INTEGER NOT NULL,
                CONSTRAINT id_pais PRIMARY KEY (id_pais)
);
COMMENT ON TABLE hr.paises IS 'Table que contém as informações relativas a países.';
COMMENT ON COLUMN hr.paises.id_pais IS 'Chave primária da tabela paises.';
COMMENT ON COLUMN hr.paises.nome IS 'Nome do país contido na tabela.';
COMMENT ON COLUMN hr.paises.id_regiao IS 'Chave estrangeira referente a tabela regiões.';


CREATE UNIQUE INDEX paises_idx
 ON hr.paises
 ( nome );

CREATE TABLE hr.localizacoes (
                id_localizacao INTEGER NOT NULL,
                cep VARCHAR(12),
                endereco VARCHAR(50),
                uf VARCHAR(25),
                cidade VARCHAR(50),
                id_pais CHAR(2) NOT NULL,
                CONSTRAINT id_localizacao PRIMARY KEY (id_localizacao)
);
COMMENT ON TABLE hr.localizacoes IS 'Tabela localizações. Contém os endereços de escritórios 
e ambientes relativos a empresa. Não contém informações de clientes.';
COMMENT ON COLUMN hr.localizacoes.id_localizacao IS 'Chave primária da tabela localizações.';
COMMENT ON COLUMN hr.localizacoes.cep IS 'CEP da localização.';
COMMENT ON COLUMN hr.localizacoes.endereco IS 'Endereço (logradouro, número) de um ambiente relacionado à empresa.';
COMMENT ON COLUMN hr.localizacoes.uf IS 'Estado ou distrito (em sigla ou por extenso) onde está localizado o ambiente relacionado à empresa.';
COMMENT ON COLUMN hr.localizacoes.cidade IS 'Nome da cidade onde está localizado o ambiente relaciona à empresa.';
COMMENT ON COLUMN hr.localizacoes.id_pais IS 'Chave estrangeira referente a tabela países.';


CREATE TABLE hr.departamentos (
                id_departamento INTEGER NOT NULL,
                nome VARCHAR(50) NOT NULL,
                id_localizacao INTEGER NOT NULL,
                id_supervisor INTEGER,
                CONSTRAINT id_departamento PRIMARY KEY (id_departamento)
);
COMMENT ON TABLE hr.departamentos IS 'Tabela com informações sobre os departamentos da empresa.';
COMMENT ON COLUMN hr.departamentos.id_departamento IS 'Chave primária da tabela departamentos.';
COMMENT ON COLUMN hr.departamentos.nome IS 'Nome do departamento contido na tabela.';
COMMENT ON COLUMN hr.departamentos.id_localizacao IS 'CHave estragngeira referente a tabela localizações.';
COMMENT ON COLUMN hr.departamentos.id_supervisor IS 'identificador do supervisor do departamento.';


CREATE UNIQUE INDEX departamentos_idx
 ON hr.departamentos
 ( nome );

CREATE TABLE hr.empregados (
                id_empregado INTEGER NOT NULL,
                nome VARCHAR(75) NOT NULL,
                email VARCHAR(35) NOT NULL,
                telefone VARCHAR(20),
                data_contratacao DATE NOT NULL,
                id_cargo VARCHAR(10) NOT NULL,
                salario NUMERIC(8,2),
                comissao NUMERIC(8,2),
                id_supervisor INTEGER,
                id_departamento INTEGER,
                CONSTRAINT id_empregado PRIMARY KEY (id_empregado)
);
COMMENT ON TABLE hr.empregados IS 'Tabela que contém as informações sobre os empregados.';
COMMENT ON COLUMN hr.empregados.id_empregado IS 'Chave primária da tabela empregados.';
COMMENT ON COLUMN hr.empregados.nome IS 'Nome completo do empregado.';
COMMENT ON COLUMN hr.empregados.email IS 'Email do empregado para contato, sendo somente a parte inicial (antes do @).';
COMMENT ON COLUMN hr.empregados.telefone IS 'Telefone do empregado para contato (há espaço para ddd e ddi).';
COMMENT ON COLUMN hr.empregados.data_contratacao IS 'Data referente ao dia que o empregado iniciou no cargo.';
COMMENT ON COLUMN hr.empregados.id_cargo IS 'Chave estrangeira referente a tabela cargos.';
COMMENT ON COLUMN hr.empregados.salario IS 'Salário mensal do empregado.';
COMMENT ON COLUMN hr.empregados.comissao IS 'Porcentagem de comissão de um empregado. Somente os empregados no departamento de vendas são elegíveis para comissão.';
COMMENT ON COLUMN hr.empregados.id_departamento IS 'Chave estrangeira referente a tabela departamentos. Informa a que departamento o empregado está atuando no momento.';
COMMENT ON COLUMN hr.empregados.id_supervisor IS 'Chave primária da tabela empregados.';


CREATE UNIQUE INDEX empregados_idx
 ON hr.empregados
 ( email );

CREATE TABLE hr.historico_cargos (
                id_empregado INTEGER NOT NULL,
                data_inicial DATE NOT NULL,
                data_final DATE NOT NULL,
                id_cargo VARCHAR(10) NOT NULL,
                id_departamento INTEGER NOT NULL,
                CONSTRAINT data_inicial PRIMARY KEY (id_empregado, data_inicial)
);
COMMENT ON TABLE hr.historico_cargos IS 'Tabela que armazena o histórico de cargos de um empregado. Caso o empregado mude de cargo no departamento, ou de departamento em um cargo, é necessário a adição de uma nova linha de informação na tabela, com as informações antigas do empregado.';
COMMENT ON COLUMN hr.historico_cargos.id_empregado IS 'Chave primária da tabela empregados.';
COMMENT ON COLUMN hr.historico_cargos.data_inicial IS 'CHave primária da tabela histórico de cargos.';
COMMENT ON COLUMN hr.historico_cargos.data_final IS 'Data final de um empregado neste cargo, esta data deve ser maior que a data inicial.';
COMMENT ON COLUMN hr.historico_cargos.id_cargo IS 'Chave estrangeira referente a tabela cargos.';
COMMENT ON COLUMN hr.historico_cargos.id_departamento IS 'Chave estrangeira referente a tabela departamentos.';
